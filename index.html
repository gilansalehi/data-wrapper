DWRL Syntax Reference

DWRL (Data Wrapper Resource Locator) is a URL-shaped addressing scheme parsed by the browser's native new URL(). Intent (read vs. write) is determined by the token ($, _, @), not a verb.

Path Resolution

<!-- Bare key or ./ prefix: item-scoped inside $list, falls back to wrapper state -->
<span $text="./task"></span>

<!-- Leading /: always wrapper-root state -->
<span $text="/username"></span>

<!-- Nested path: drills into nested objects -->
<span $text="/user/profile/name"></span>

<!-- Protocol-relative //: targets a specific wrapper by id -->
<span $text="//cart/total"></span>

Text & Property Bindings ($)

<span $text="/username"></span>
<div  $html="/bio"></div>
<input $value="/draft">
<input type="checkbox" $checked="/settings/notifications">
<button $disabled="/isLoading">Submit</button>
<a $href="/user/profileUrl">Profile</a>
<img $src="/user/avatarUrl" $alt="/user/name">
<input $placeholder="/ui/searchHint">

Formatters via Query Params

<span $text="/price?format=currency"></span>
<span $text="/label?format=upper"></span>
<span $text="/items?format=count"></span>

<!-- Chained: applied left to right -->
<span $text="/name?format=trim&format=upper"></span>

<!-- Locale-aware (proposed) -->
<span $text="/createdAt?format=date&locale=en-GB"></span>
<span $text="/price?format=currency&currency=EUR&locale=de-DE"></span>

<!-- Boolean formatters -->
<span $text="/isActive?format=yesno"></span>
<span $class="/isActive?format=onoff"></span>

Class Bindings ($class)

<!-- Value-as-class: state value used as class string. Static class="" preserved. -->
<li class="todo-item" $class="./status"></li>
<!-- state.status = "active" → class="todo-item active" -->

<!-- toggle=name: add/remove class based on truthiness (proposed) -->
<li class="todo-item" $class="./isDone?toggle=done"></li>

<!-- map=val:class,...: map discrete values to class names (proposed) -->
<li $class="./status?map=active:is-active,done:is-done"></li>

<!-- prefix: prepend a namespace to the value (proposed) -->
<div class="card" $class="/theme?prefix=theme-"></div>

<!-- Boolean formatters for class -->
<span class="badge" $class="/isActive?format=onoff"></span>
<!-- true → class="badge on", false → class="badge off" -->

Event Wiring (@)

<button @click="todo/remove" $value="./id">Delete</button>
<form   @submit="todo/add">...</form>
<input  @input="search/update">
<button @click="filter/set" data-val="active">Active</button>

<!-- Handler receives CustomEvent; original event is in e.detail -->
<script>
wrapper.register({
    'todo/add': e => { e.detail.preventDefault(); ... },
    'filter/set': e => { app.put('filter', e.detail.delegateTarget.dataset.val); },
});
</script>

List Rendering ($list)

<ul $list="/todos">
    <template>
        <li $text="./task"></li>
    </template>
</ul>

<!-- Empty-state template -->
<template id="no-results"><li class="empty">Nothing here.</li></template>
<ul $list="/results" data-empty="no-results">
    <template><li $text="./label"></li></template>
</ul>

<!-- Nested list: item-scoped $list (proposed) -->
<ul $list="/projects">
    <template>
        <li>
            <strong $text="./name"></strong>
            <ul $list="./tasks">
                <template><li $text="./title"></li></template>
            </ul>
        </li>
    </template>
</ul>

<!-- Filtered list via query param (proposed) -->
<ul $list="/todos?where=done:false">
    <template><li $text="./task"></li></template>
</ul>

Cross-Component Mesh

<data-wrapper id="cart">
    <span $text="/items?format=count"></span>
    <span $text="/total?format=currency"></span>
</data-wrapper>

<data-wrapper id="header">
    <span $text="//cart/items?format=count"></span>
    <button @click="action://cart/open">Cart</button>
</data-wrapper>

Component Init (onload + register)

<data-wrapper id="app" data-filter="all"
    onload="this.register({
        'todo/add':    e => { ... },
        'todo/remove': e => { ... }
    })">
</data-wrapper>

<!-- Or from a module script -->
<script type="module">
    await customElements.whenDefined('data-wrapper');
    document.getElementById('app').register({ 'todo/add': e => { ... } });
</script>
